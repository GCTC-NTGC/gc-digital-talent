# Scroll Area

> Augments native scroll functionality for custom, cross-browser styling.

This is a styled version of [@radix-ui/scroll-area](https://www.radix-ui.com/docs/primitives/components/scroll-area).

## Subcomponents

- `ScrollArea.Root`
- `ScrollArea.Viewport`
- `ScrollArea.Scrollbar`
- `ScrollArea.Thumb`
- `ScrollArea.Corner`

## Usage

<Story id="components-scrollarea--default" />

> **Note**
> It is important to set a height and width value on `ScrollArea.Root`.

```tsx
import ScrollArea from "@gc-digital-talent/ui/ScrollArea";

const ScrollAreaExample = () => {
  const [value, setValue] = React.useState<string>("");
  return (
    <ScrollArea.Root style={{ height: 150, width: 320 }}>
      <ScrollArea.Viewport data-h2-background-color="base(white)">
        <div data-h2-padding="base(x.25, x.5)">
          {ITEMS.map((item) => (
            <p key={item}>{item}</p>
          ))}
        </div>
      </ScrollArea.Viewport>
      <ScrollArea.Scrollbar orientation="vertical">
        <ScrollArea.Thumb />
      </ScrollArea.Scrollbar>
    </ScrollArea.Root>
  );
};
```
